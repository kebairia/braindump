#+title: record using ffmpeg
#+filetags: :video:unix:
#+begin_src sh
  ffmpeg -f x11grab -s 1366x768 -i :0.0 out.mkv
#+end_src
- -f :: format, here it *grab* your x11
- -s :: size, set the resolution of the video
- -i :: input, from where it get the data, here 0.0 means
  you main monitor
