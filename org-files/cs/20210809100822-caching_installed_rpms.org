#+title: caching installed rpms
#+filetags: :dnf:linux:redhat:
#+begin_src sh
  echo "keepcache=1" >> /etc/dnf/dnf.conf
#+end_src
then visit ~/var/cache/dnf~
if you want to clean all the downloaded packages afterward, use:
#+begin_src sh
  sudo dnf clean packages
#+end_src
